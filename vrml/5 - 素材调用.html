5 - 素材调用

组节点：Group, Transform, Inline, Anchor, Switch, Billboard, LOD (Level of Detail)

采用 DEF 节点定义、USE 节点引用

DEF 节点别名 节点 {
# ...
}
USE 节点别名
仅在本文件内有效。

【例】画米字 Cylinder radius 0.1 height 4

Inline 节点

调用外部素材文件，无法修改
Inline {
url ""
bboxCenter 0 0 0
bboxSize -1 -1 -1
}
【例】把钟表场景 (clock.wrl) 和 "Return!" 文字 (return.wrl) 组合在一起，并把 "Return!" 文字放在钟表的正下方。
Inline {
url "clock.wrl"
}
Transform { # 一般 Inline 会和 Transform 配合使用，避免多场景重叠
translation 0 -3 0
children [
Inline {
url "return.wrl"
}
]
}

Anchor 节点

跳转到指定场景，取代当前场景（类似于超链接）
Anchor {
children [] # 用作锚的造型
url "" # 跳转目标（外部 .wrl 文件）
description "" # 鼠标放上去时的提示信息
}
【例】点击 "Go!" 文字跳转到 inline.wrl 场景。
Anchor {
children [
Shape {
geometry Text {
string "Go!"
}
}
]
url "inline.wrl"
}

使用 PROTO 定义节点类型和 EXTERNPROTO

调用后可更改。
PROTO <NodeTypeName>
[
# 接口域部分
field <fieldType> <fieldName> <defaultValue>
exposedField <fieldType> <fieldName> <defaultValue>
eventIn <eventInType> <eventInName>
eventOut <eventOutType> <eventOutName>
]
{
# 节点体：定义节点执行内容
<nodebody>
}
【例】使用PROTO语句定义一个名为Coclock的钟表原型节点，并为其设置表身颜色、表身尺寸及钟表位置三个接口域；调用原型节点时，通过设置接口域的域值，改变钟表的位置、颜色及尺寸。
#VRML V2.0 utf8
PROTO Coclock
[
field SFColor bodyColor 0 1 0
field SFVec3f bodySize 1 1 1
field SFVec3f bodyPos 0 0 0
]
{
Transform {
scale IS bodySize
translation IS bodyPos
children [
Shape { # Clock case
appearance Appearance {
material Material {
diffuseColor IS bodyColor
}
}
geometry Box {
size 4 4 1
}
}
Transform { # Lift up the clock surface
translation 0 0 0.6
children [
DEF c Shape { # Middle circle
appearance Appearance {
material Material {
diffuseColor 1 1 0
}
}
geometry Sphere {
radius 0.1
}
}
Transform { # minute hand
translation 0 0.95 0
children [
Shape {
appearance Appearance {
material Material {
diffuseColor 1 0 0
}
}
geometry Cylinder {
radius 0.1
height 1.7
}
}
]
}
Transform { # hour hand
translation 0.85 0 0
rotation 0 0 1 1.57
children [
Shape {
appearance Appearance {
material Material {
diffuseColor 1 0 0
}
}
geometry Cylinder {
radius 0.1
height 1.5
}
}
]
}
DEF qd Transform { # 10, 11, 12
translation 0 1.9 0
children [
USE c
Transform {
center 0 -1.9 0
rotation 0 0 1 0.523
children [
USE c
Transform {
center 0 -1.9 0
rotation 0 0 1 0.523
children [
USE c
]
}
]
}
]
}
Transform { # 7, 8, 9
rotation 0 0 1 1.57
children [
USE qd
]
}
Transform { # 4, 5, 6
rotation 0 0 1 3.14
children [
USE qd
]
}
Transform { # 1, 2, 3
rotation 0 0 1 -1.57
children [
USE qd
]
}
]
}
]
}
}
# 调用原型
Coclock {
bodySize 0.5 0.5 0.5
}
EXTERNPROTO <NodeTypeName>
[
# 接口域部分
field <fieldType> <fieldName> # 没有默认值了 可控域需要抄过来，不抄的不可更改
exposedField <fieldType> <fieldName> # 没有默认值了
eventIn <eventInType> <eventInName>
eventOut <eventOutType> <eventOutName>
]
url / urlList / (省略) "filename.wrl#PROTO节点名"
【例】利用原型节点定义一个节点Desk，并为其设置deskTopColor, deskLegColor, deskPosition三个接口域。在调用原型节点时，通过改变接口域值，可以设置桌子位置，桌面颜色及桌腿颜色。
#VRML V2.0 utf8
PROTO Desk
[
field SFVec3f deskPos 0 0 0 # 位置
field SFColor topColor 0 1 0 # 桌面颜色
field SFColor legColor 1 1 0 # 桌腿颜色
field SFVec3f deskSize 1 1 1 # 桌子大小（用于 scale）
field SFRotation deskOrientation 0 0 1 0 # 桌子朝向
]
{
Transform {
scale IS deskSize
translation IS deskPos
rotation IS deskOrientation
children [
Shape { # 桌面
appearance Appearance {
material Material {
diffuseColor IS topColor
}
}
geometry Box {
size 4 0.1 2
}
}
Transform { # 桌腿1（左上角）
translation -1.5 -0.5 0.8
children [
DEF leg Shape {
appearance Appearance {
material Material {
diffuseColor IS legColor
}
}
geometry Cylinder {
height 1
radius 0.1
}
}
]
}
Transform { # 桌腿2（左下角）
translation -1.5 -0.5 -0.8
children [
USE leg
]
}
Transform { # 桌腿3（右下角）
translation 1.5 -0.5 -0.8
children [
USE leg
]
}
Transform { # 桌腿4（右上角）
translation 1.5 -0.5 0.8
children [
USE leg
]
}
]
}
}
#VRML V2.0 utf8
EXTERNPROTO Desk
[
field SFVec3f deskPos
field SFColor topColor
field SFColor legColor
field SFVec3f deskSize
field SFRotation deskOrientation
]
"Desk.wrl" # 或 "Desk.wrl#Desk"
# 调用 Desk
Desk {}
